import { Component, Input, OnInit } from '@angular/core';
import { FormGroup } from '@angular/forms';

@Component({
  selector: 'safe-tab-filter',
  templateUrl: './tab-filter.component.html',
  styleUrls: ['./tab-filter.component.scss']
})
export class SafeTabFilterComponent implements OnInit {

  @Input() form: FormGroup = new FormGroup({});
  @Input() filters: any[] = [];

  private inputs = '';

  constructor() { }

  ngOnInit(): void {
  }

  setCurrentDate(filterName: string): void {
    this.form.controls[filterName].setValue('today()');
  }

  onKey(e: any, filterName: string): void {
    if (e.target.value === '') { this.inputs = ''; }
    if (e.keyCode === 8) {
      this.inputs = this.inputs.slice(0, this.inputs.length - 1);
    }
    if (this.inputs.length <= 9) {
      if (RegExp('^[0-9]*$').test(e.key)) {
        if (this.inputs.length === 3 || this.inputs.length === 6) {
          this.inputs += e.key + '/';
          e.target.value += '/';
        } else {
          this.inputs += e.key;
        }
      } else if (e.key === '/') {
        e.stopPropagation();
        this.inputs = this.inputs.slice(0, this.inputs.length - 1);
      } else {
        e.stopPropagation();
        e.target.value = e.target.value.replace(/[^0-9\/]/g, '');
      }
    } else {
      e.stopPropagation();
      e.target.value = this.inputs;
    }
    if (this.inputs.length > 9 && !RegExp('\\d{4}\\/(0?[1-9]|1[012])\\/(0?[1-9]|[12][0-9]|3[01])*').test(this.inputs)) {
      this.form.controls[filterName].setErrors({incorrect: true});
    }
  }
}

<form [formGroup]="form">
    <ng-container *ngFor="let filter of filters">
        <mat-form-field appearance="outline">
            <mat-label>{{filter.name}}</mat-label>
            <ng-container *ngIf="filter.type.name === 'Int'">
                <input matInput [formControlName]="filter.name" type="number" />
            </ng-container>
            <ng-container *ngIf="filter.type.name === 'Boolean'">
                <mat-select [formControlName]="filter.name">
                    <mat-option>-- None --</mat-option>
                    <mat-option [value]="true">True</mat-option>
                    <mat-option [value]="false">False</mat-option>
                </mat-select>
            </ng-container>
            <ng-container
                *ngIf="!['DateTime', 'Date', 'Int', 'Boolean'].includes(filter.type.name)">
                <input matInput [formControlName]="filter.name" type="string" />
            </ng-container>
            <ng-container *ngIf="['Date', 'DateTime'].includes(filter.type.name)">
                <input matInput matTooltip="First enter the year then month and day" [matTooltipPosition]="'above'"
                    maxlength="10" [formControlName]="filter.name" type="text" (keyup)="onKey($event, filter.name)" />
            </ng-container>
            <mat-icon *ngIf="['Date', 'DateTime'].includes(filter.type.name)" color="primary" matSuffix matTooltip="Always update to today's date"
                [matTooltipPosition]="'above'" (click)="setCurrentDate(filter.name)">calendar_today</mat-icon>
        </mat-form-field>
    </ng-container>
</form>
